### 第 k 个排列

::: tip 难度
中等
:::

![20200905](http://qiniu.gaowenju.com/leecode/banner/20200905.jpg)

## [题目:](https://leetcode-cn.com/problems/permutation-sequence/)

给出集合 [1,2,3,…,n]，其所有元素共有 n! 种排列。

按大小顺序列出所有排列情况，并一一标记，当 n = 3 时, 所有排列如下：

1. "123"
2. "132"
3. "213"
4. "231"
5. "312"
6. "321"
   给定 n 和 k，返回第 k 个排列。

### 说明：

- 给定 n 的范围是 [1, 9]。
- 给定 k 的范围是[1, n!]。

### 示例:

1. 示例 1

```
输入: n = 3, k = 3
输出: "213"
```

2. 示例 2

```
输入: n = 4, k = 9
输出: "2314"
```

## 抛砖引玉

![抛砖引玉](http://qiniu.gaowenju.com/leecode/20200905.png)

**思路**

正序转换成倒序，求中间第 k 个排列，那么首先想到的就是这个排序的过程是什么样的：

一般思考排序最先想到的就是逐位排序，如果后一位大于前一位则互换：

1. 从前向后
2. 从后向前

但是本题的示例给的方式并不是上面方式中的一种。

---

那么看下示例中排序的顺序是什么样的：

1. 123、132，都是以 1 开始
2. 213、231，都是以 2 开始
3. 312、321，都是以 3 开始

依次将 n 中从小到大的元素作为开始枚举排序，题目中：所有元素共有 n! 种排列（n=3，3\*2\*1=6）：

- 其中 1 开始有(n-1)!中（2\*1=2）第 2
- 其中 2 开始有(n-1)!中（2\*1=2）第 4
- 其中 3 开始有(n-1)!中（2\*1=2）第 6

那么当求第 k 个时就可以直接定位到是哪个数字开始的。

继续推导，同样的逻辑将第二位的数字作为起始位看：

其中 1 开始时:

- 其中 2 开始 1，第 1
- 其中 3 开始 1，第 2
  ....

那么，换个方式理解下，已知一个排序结果，231：

- 根据第一个字符 2 就已知其排序范围应该是第 4 到第 6
- 根据第二个字符 3 就已知其排序范围应该是第 5

**实现**

从前到后逐个确定字符

生成字符数组字符对应的阶乘数组：

- nums：[1,2,3]
- list: [2,4,6]

- 找到第一个大于 k 的阶乘结果，则确定第一个字符未当前这个索引前一个字符
- 继续确定第二个字符

```javascript
    let num = 1

      num = num * i
    }
    return num
  }
  // 传入固定数组x，返回index，使inde想的阶乘结果和小于等于x
  function getIndex(x) {
    let num = 1,
      next = 1,
      val = 1
    while (next < x) {
      next
    }
    for (let i = 1; i <= n; i++) {

    }
    return num
  }
}
```

```javascript
/**
 * @param {number} n
 * @param {number} k
 * @return {string}
 */
var getPermutation = function (n, k) {
  let _result = '',
    nums = [],
    list = [],
    val = 1,
    index = 0
  for (let i = 1; i <= n; i++) {
    nums.push(i)
    if (i !== n) {
      val = val * i
      list.push(val)
    }
  }

  while (list.length > 0) {
    let len = list.length
    if (k > list[len - 1]) {
      k = k - list[len - 1]
      index++
    } else {
      _result += nums.splice(index, 1)[0]
      index = 0
      list.pop()
    }
  }
  return _result + nums[0]
}
```

```javascript
/**
 * @param {number} n
 * @param {number} k
 * @return {string}
 */
var getPermutation = function (n, k) {
  let num = [] // ['1','2','3','4']
  let fa = []
  let tmp = 1
  for (let i = 1; i <= n; i++) {
    num.push(String(i)) // 数字符集合
    tmp *= i
    fa.push(tmp) // 阶乘集合
  }
  fa.pop() // 最后一个是总数，没有作用所以推出它
  let count = 0 // 当前选中字符位置
  let res = ''
  while (fa.length > 0) {
    let sub = fa.length - 1
    if (k > fa[sub]) {
      // 大于就减去并位置+1
      k = k - fa[sub]
      count++
    } else {
      // 小于等于就输出到字符串，重置字符位置
      res += num.splice(count, 1)[0]
      count = 0
      fa.pop()
    }
  }
  return res + num[0] // 遍历完还剩一个在数组中，加上
}
```

```javascript
/**
 * @param {number} n
 * @param {number} k
 * @return {string}
 */
var getPermutation = function (n, k) {
  let count = 0
  const used = new Set()

  const helper = (temp) => {
    if (temp.length == n) {
      // 选齐了 生成了一个排列
      count++
      if (count == k) {
        // 正好是第k个
        return temp.join('') // 拼成字符串，返回出来
      }
      return // 结束当前递归，考察别的选择
    }
    for (let i = 1; i <= n; i++) {
      // 枚举出所有选择
      if (used.has(i)) continue // 已经选过，跳过
      temp.push(i) // 选择
      used.add(i)
      const res = helper(temp) // 递归 往下选，获取返回值
      temp.pop() // 撤销选择
      used.delete(i)
      if (res) {
        // 有返回值，说明找到了，返回res，结束掉遍历
        return res
      }
    }
  }

  return helper([])
}
```

```javascript
/**
 * @param {number} n
 * @param {number} k
 * @return {string}
 */
var getPermutation = function (n, k) {
  let factorial = 1,
    _result = '',
    valid = []
  for (let i = 1; i <= n; i++) {
    valid.push(i) // [1, 2, 3, 4]
    factorial = factorial * i // 4!   24
  }
  k--
  while (valid.length > 0) {
    // 选了一个数字就删掉，直到为空
    factorial = parseInt(factorial / valid.length, 10)
    const index = parseInt((k / factorial) | 0, 10)
    _result += valid[index] // 加上当前选的数字
    valid.splice(index, 1) // nums删去选的这个数字
    k = k % factorial // 更新 k，
  }
  return _result
}
```
